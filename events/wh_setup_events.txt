	hide_window = yes

			OR = {
				is_republic = yes

			}
		}

			dynasty = none
		}
		culture_group = bretonnian_group
		OR = {
			religion = grail
			religion = false_grail
		}
	}

	immediate = {
		log = "work event DR_startup.4"
		if = {
			limit = {

						trait = bretonnian_knight

					martial = 10

			}

				give_minor_title = title_sir_bretonnian
			}
		}
		if = {
			limit = {
				OR = {
					trait = bretonnian_knight

					has_minor_title = title_sir_bretonnian
