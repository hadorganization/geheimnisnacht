						remove_trait = dim

						remove_trait = slow
						remove_trait = idiot
						remove_trait = imbecile

						change_intrigue = 1

						id = fickletzeentch.1
					}
				}
			}
			25 = {
				if = {
					limit = {
						nor = {
							trait = mut_thirdeye
							trait = mutation_minor
							trait = dumb
						}
					}
					add_trait = mutation_minor
					add_trait = dumb
					set_character_flag = mut_thirdeye
				}
				hidden_tooltip = {
					change_variable = {
						which = "mutation_counter" value = 1
					}

						id = mutation.2
					}

					limit = {

						remove_trait = dim
						remove_trait = dumb
						remove_trait = slow
						remove_trait = idiot
						remove_trait = imbecile

					character_event = {
						id = fickletzeentch.1
					}
					change_learning = 2
					change_intrigue = 2
				}
			}
			45 = {
				if = {
					limit = {
						nor = {
							trait = mut_tongue
							trait = mutation_major
							trait = imbecile
							trait = disfigured
						}
					}
					add_trait = disfigured
					add_trait = imbecile
					add_trait = mutation_major
					set_character_flag = mut_tongue
					hidden_tooltip = {
						remove_trait = genius

						remove_trait = quick
						remove_trait = smart
