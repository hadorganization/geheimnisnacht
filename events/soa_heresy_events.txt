# Heresy Appears in Province

	id = SoA.4000
	desc = EVTDESC39000
	picture = GFX_evt_heretic
	border = GFX_event_normal_frame_religion
	is_triggered_only = yes

	trigger = {
		# has_dlc = "Sons of Abraham"
		OR = {
			religion_group = old_world_group
			religion_group = ormazd_gods_group
		}
		# These will get Shallyan uprisings instead
		NOR = {
			religion = grail
			religion = false_grail
		}
		num_of_settlements = 1
		any_province_holding = {
			NOT = {
				holding_type = nomad
			}
		}
		NOT = {
			religion_authority = 0.9
		}
	}

	weight_multiplier = {
		factor = 0.01250
		modifier = {
			factor = 0.67
			religion_authority = 0.6
		}
		modifier = {
			factor = 0.67
			religion_authority = 0.7
		}
		modifier = {
			factor = 0.67
			religion_authority = 0.8
		}
		modifier = {
			factor = 0.50
			religion_authority = 0.9
		}
		# modifier = {
		# factor = 3.0
		# OR = {
		# province_id = 333 # Rome
		# province_id = 497 # Thrace
		# province_id = 719 # Mecca
		# }
		# }
		modifier = {
			factor = 2.00
			NOT = {
				religion_authority = 0.2
			}
		}
		modifier = {
			factor = 1.33
			NOT = {
				religion_authority = 0.3
			}
		}
		modifier = {
			factor = 1.33
			NOT = {
				religion_authority = 0.4
			}
		}
		modifier = {
			factor = 1.33
			NOT = {
				religion_authority = 0.5
			}
		}
		modifier = {
			factor = 2.00
			any_neighbor_province = {
				is_land = yes
				has_province_modifier = fanatic_uprising
			}
		}
		modifier = {
			factor = 4.00
			any_neighbor_province = {
				is_land = yes
				is_heresy_of = ROOT
				has_province_modifier = heretic_stronghold
			}
		}
	}

	immediate = {
		log = "work event id = SoA.4000 for [Root.GetBestName]"
		# # Two years of much higher revolt risk
		# add_province_modifier = {
		# name = religious_unrest
		# duration = 730
		# }
		add_province_modifier = {
			name = fanatic_uprising # Ten years of even higher revolt risk
			duration = 1825
		}
		# One heretic province in the realm is the epicentre of the heresy
		if = {
			limit = {
				owner = {
					top_liege = {
						NOT = {
							any_realm_province = {
								religion = ROOT
								has_province_modifier = heretic_stronghold
							}
						}
					}
				}
			}
			add_province_modifier = {
				name = heretic_stronghold # Ten years of even higher revolt risk
				duration = 3650
			}
		}
		# Notify the lieges
		owner = {
			any_liege = {
				character_event = {
					id = 39400
				}
			}
		}
		owner = {
			if = {
				limit = {
					ai = no
				}
				chronicle = {
					entry = CHRONICLE_HERESY_SPREAD
					picture = GFX_evt_heretic
				}
			}
		}
		owner = {
			any_liege = {
				if = {
					limit = {
						ai = no
					}
					chronicle = {
						entry = CHRONICLE_HERESY_SPREAD
						picture = GFX_evt_heretic
					}
				}
			}
		}
	}

	option = {
		name = EVTOPTA39000 # The owner is of the parent religion
		# trigger = { owner = { is_parent_religion = ROOT } }
		owner = {
			piety = -10
		}
	}
}
