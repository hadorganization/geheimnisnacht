# demon avoids execution
character_event = {
	id = wh_chaos_demons.1
	desc = EVTDESCsdemons1
	picture = GFX_evt_tzeentch
	is_triggered_only = yes

	immediate = {
		log = "work event id = wh_chaos_demons.1 for [Root.GetBestName]"
		FROMFROM = {
			prisoner = no
			banish = yes
			add_rival = ROOT
		}
		if = {
			limit = {
				any_lover = {
					character = FROMFROM
				}
			}
			remove_lover = FROMFROM
		}
	}

	option = {
		name = EVTOPTAsdemons1
	}
}

# free demon kills random person
character_event = {
	id = wh_chaos_demons.3
	desc = EVTDESCsdemons3
	picture = GFX_evt_tzeentch
	capable_only = yes
	religion_group = chaos_gods_group

	trigger = {
		is_demon_trigger = yes
		NOT = {
			num_of_friends = 1
		}
	}

	mean_time_to_happen = {
		years = 4
		modifier = {
			factor = 0.75
			trait = khorne_demon

		}
		modifier = {
			factor = 0.85
			trait = tzeentch_demon

		}
		modifier = {
			factor = 0.95
			trait = slaanesh_demon

		}
		modifier = {
			factor = 1.25
			trait = nurgle_demon

		}
	}

	immediate = {
		log = "work event id = wh_chaos_demons.3 for [Root.GetBestName]"
	}

	option = {
		name = EVTOPTAsdemons3
		random_courtier = {
			limit = {
				is_ruler = no
				is_demon_trigger = no
				nor = {
					trait = creature_nehekharan
					trait = creature_slann
					trait = everqueen
					trait = malekith
					trait = avatar_kurnous
					trait = avatar_isha
					has_artifact = carstein_ring
					wh_is_mage_or_apprentice_trigger = yes
				}
			}
			if = {
				limit = {
					is_invincible_trigger = yes
				}
				antideath_effect = yes
				break = yes
			}
			death = {
				death_reason = death_murder
				killer = ROOT
			}
		}
		if = {
			limit = {
				liege = {
					NOT = {
						religion_group = chaos_gods_group
					}
				}
			}
			liege = {
				opinion = {
					modifier = opinion_demon_havoc
					who = ROOT
				}
			}
		}
	}
}

# Demon adds province modifier wherever he is
character_event = {
	id = wh_chaos_demons.8
	desc = EVTDESCsdemons8
	picture = GFX_evt_tzeentch
	only_rulers = yes
	capable_only = yes
	religion_group = chaos_gods_group

	trigger = {
		is_demon_trigger = yes
	}

	mean_time_to_happen = {
		years = 4
		modifier = {
			factor = 0.75
			trait = khorne_demon

		}
		modifier = {
			factor = 0.85
			trait = tzeentch_demon

		}
		modifier = {
			factor = 0.95
			trait = slaanesh_demon

		}
		modifier = {
			factor = 0.95
			trait = nurgle_demon

		}
	}

	immediate = {
		log = "work event id = wh_chaos_demons.8 for [Root.GetBestName]"
	}

	option = {
		name = EVTOPTAsdemons8
		if = {
			limit = {
				trait = khorne_demon
			}
			location = {
				add_province_modifier = {
					name = beastmen_unrest
					duration = 4000
				}
			}
		}
		if = {
			limit = {
				or = {
					trait = tzeentch_demon
					trait = hashut_kdaai
				}
			}
			location = {
				add_province_modifier = {
					name = warpdust_traces
					duration = 4000
				}
			}
		}
		if = {
			limit = {
				trait = slaanesh_demon
			}
			location = {
				add_province_modifier = {
					name = slaanesh_hedonism
					duration = 4000
				}
			}
		}
		if = {
			limit = {
				trait = nurgle_demon
			}
			location = {
				add_province_modifier = {
					name = nurgle_province_disease
					duration = 4000
				}
			}
			random_list = {
				10 = {
					location = {
						spawn_disease = tuberculosis
					}
				}
				10 = {
					location = {
						spawn_disease = typhoid_fever
					}
				}
				10 = {
					location = {
						spawn_disease = typhus
					}
				}
				10 = {
					location = {
						spawn_disease = measles
					}
				}
				10 = {
					location = {
						spawn_disease = small_pox
					}
				}
				10 = {
					location = {
						spawn_disease = syphilis
					}
				}
			}
		}
	}
}
