# demon avoids execution
character_event = {
	id = sdemons.1
	desc = EVTDESCsdemons1
	picture = GFX_evt_tzeentch
	is_triggered_only = yes

	immediate = {
		FROMFROM = {
			prisoner = no
			banish = yes
			add_rival = ROOT
		}
		if = {
			limit = {
				any_lover = {
					character = FROMFROM
				}
			}
			remove_lover = FROMFROM
		}
		if = {
			limit = {
				any_rival = {
					character = FROMFROM
				}
			}
			remove_lover = FROMFROM
		}
	}

	option = {
		name = EVTOPTAsdemons1
	}
}
