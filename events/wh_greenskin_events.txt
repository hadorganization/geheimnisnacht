# Now to assign the size traits for orcs that don't have them
character_event = {
	id = orcs.1
	title = YOUR_SIZE
	desc = EVTDESCORCS.1 # Orcs and goblins grow in size from the number of battles they have fought in and won. As you have reached the age of adulthood, your size is...
	picture = "GFX_evt_orcs"
	is_triggered_only = yes

	trigger = {
		age = 16
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		NOT = {
			trait = orc_small
		}
		NOT = {
			trait = orc_average
		}
		NOT = {
			trait = orc_large
		}
		NOT = {
			trait = orc_huge
		}
		NOT = {
			trait = goblin_small
		}
		NOT = {
			trait = goblin_average
		}
		NOT = {
			trait = goblin_large
		}
		NOT = {
			trait = goblin_huge
		}
	}

	option = {
		name = ARRRRGH
		if = {
			limit = {
				culture_group = orc_group
				martial = 5
				NOT = {
					martial = 10
				}
			}
			add_trait = orc_small
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 10
				NOT = {
					martial = 15
				}
			}
			add_trait = orc_average
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 15
				NOT = {
					martial = 25
				}
			}
			add_trait = orc_large
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 25
			}
			add_trait = orc_huge
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 5
				NOT = {
					martial = 10
				}
			}
			add_trait = goblin_small
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 10
				NOT = {
					martial = 15
				}
			}
			add_trait = goblin_average
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 15
				NOT = {
					martial = 25
				}
			}
			add_trait = goblin_large
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 25
			}
			add_trait = goblin_huge
		}
	}
}

# Event for greenskins to grow in size from battles
character_event = {
	id = orcs.2
	desc = EVTDESCORCS.2 # With each victory in battle, you grow a little in size and strength!
	picture = "GFX_evt_orcs"
	is_triggered_only = yes

	trigger = {
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		NOT = {
			trait = orc_huge
		}
		NOT = {
			trait = goblin_huge
		}
	}
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 2
			trait = brave
		}
		modifier = {
			factor = 2
			trait = cruel
		}
		modifier = {
			factor = 2
			trait = wroth
		}
	}

	option = {
		name = "EVTOPTAORCSGROW2"
		if = {
			limit = {
				trait = orc_small
			}
			add_trait = orc_average
			remove_trait = orc_small
		}
		if = {
			limit = {
				trait = orc_average
			}
			add_trait = orc_large
			remove_trait = orc_average
		}
		if = {
			limit = {
				trait = orc_large
			}
			add_trait = orc_huge
			remove_trait = orc_large
		}
		if = {
			limit = {
				trait = goblin_small
			}
			add_trait = goblin_average
			remove_trait = goblin_small
		}
		if = {
			limit = {
				trait = goblin_average
			}
			add_trait = goblin_large
			remove_trait = goblin_average
		}
		if = {
			limit = {
				trait = goblin_large
			}
			add_trait = goblin_huge
			remove_trait = goblin_large
		}
	}
}

# Ork War Event - Shaman spawns (Dynasty).
character_event = {
	id = orcs.13
	desc = EVTDESCORCS.13 # In times of war, It's not uncommon for Ork Shamans to join a Warboss's retinue, so that they may witness their nascent legend
	picture = "GFX_evt_orcs"
	min_age = 16
	is_triggered_only = yes

	trigger = {
		war = yes
		prisoner = no
		is_landed = yes
		culture_group = orc_group
	}

	option = {
		name = EVTDESCORCS13A # "Wot? Jus' tell 'im ta stay outta mah way!!
		prestige = 50
		piety = 25
		create_random_priest = {
			random_traits = yes
			dynasty = ROOT
			culture = ROOT
			religion = ROOT
			female = no
			age = 30
		}
		new_character = {
			change_learning = 5
			add_trait = greenskin_shaman
			z_magic_power_sequential_addition = yes
		}
		recalc_succession = yes
	}

	option = {
		name = EVTDESCORCS13B # Oi'm not sure 'bout dat... Wot if he makes meeb head splode?
		prestige = -50
		random = {
			chance = 25
			add_trait = craven
		}
	}
}
