# study mechanics menu
character_event = {
	id = studymechanics.0
	title = studymechanicstitle
	desc = studymechanics0 # While the realm is at peace, you are free to study your machinery undisturbed. Do you wish to seclude yourself and look into how things tick?
	picture = "GFX_evt_lab"

	is_triggered_only = yes

	}

	option = {
		name = YES
		add_trait = studying_mechanics
	}

	option = {
		trigger = {
			ai = no
		}
		name = NO
	}
}

# Those with the study trait gain learning over time, depending on certain modifiers
character_event = {
	id = studymechanics.1
	title = studymechanicstitle
	desc = studymechanics1 # Your studies have paid off, and you now understand a little more of what makes things tick.
	picture = "GFX_evt_lab"
	min_age = 16
	is_triggered_only = yes

	trigger = {
		trait = studying_mechanics
		trait = dwarf_engineer
		NOT = {
			learning = 25
		}
	}

	immediate = {
		log = "work event id = studymechanics.1"
	}

	option = {
		name = EXCELLENT
		change_learning = 1
	}
}

# When the Realm is at war, wizards stop their studies and join in the defence
character_event = {
	id = studymechanics.2
	title = studymechanicstitle
	desc = studymechanics2 # The realm is at war, and you are forced to stop your studies in seclusion and do what you can for it's defence.
	picture = "GFX_evt_lab"
	min_age = 16

	trigger = {
		trait = studying_mechanics
		trait = dwarf_engineer
		NOT = {
			has_character_flag = refusedtostopstudying
		}
		OR = {
			war = yes
			liege = {
				war = yes
			}
		}
	}

	mean_time_to_happen = {
		days = 7
	}

	immediate = {
		log = "work event id = studymechanics.2"
	}

	option = {
		name = OK
		remove_trait = studying_mechanics
	}

	option = {
		name = studymagic2B # No, I cannot stop my studies now!
		trigger = {
			ai = no
		}
		set_character_flag = refusedtostopstudying
		prestige = -250
	}
}

# Becoming a Master Engineer
