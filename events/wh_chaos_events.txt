# Hellgate opens
province_event = {
	id = wh_chaos.6
	desc = EVTDESCschaos6
	picture = GFX_evt_chaos_hordes

	trigger = {
		empire = {
			title = e_chaos_wastes
		}
		NOT = {
			has_province_modifier = warphole_epicentre
		}
		any_neighbor_province = {
			NOT = {
				has_province_modifier = warphole_epicentre
			}
			any_neighbor_province = {
				NOT = {
					has_province_modifier = warphole_epicentre
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 5950
	}

	immediate = {
		owner = {
			if = {
				limit = {
					religion = tzeentch
				}
				change_variable = {
					which = "tzeentch_favor"
					value = 2
				}
			}
			if = {
				limit = {
					religion = slaanesh
				}
				change_variable = {
					which = "slaanesh_favor"
					value = 2
				}
			}
			if = {
				limit = {
					religion = khorne
				}
				change_variable = {
					which = "khorne_favor"
					value = 2
				}
			}
			if = {
				limit = {
					religion = nurgle
				}
				change_variable = {
					which = "nurgle_favor"
					value = 2
				}
			}
		}
	}

	option = {
		name = EVTOPTAschaos6
		hidden_tooltip = {
			# the eye strikes
			province_event = {
				id = wh_chaos.8
				days = 25
				random = 40
			}
			# remove it
			province_event = {
				id = wh_chaos.7
				days = 250
				random = 900
			}
			add_province_modifier = {
				name = warphole_epicentre
				duration = -1
			}
			any_neighbor_province = {
				add_province_modifier = {
					name = warphole_epicentre
					duration = -1
				}
				province_event = {
					id = wh_chaos.8
					days = 25
					random = 40
				}
				province_event = {
					id = wh_chaos.7
					days = 250
					random = 900
				}
			}
		}
	}
}
