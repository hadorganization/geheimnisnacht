# Gud n' Propa Greenskin stuff
# Greenskin Power law

# The Greenskin Law has a MMTH of automatically increasing or decreasing every 1 and a half years. This time can vary depending on character traits.
# It can also be manually increased in through a decision, but you will be challenged to a duel by another Greenskin if you do so, for trying to impose your rule.

# Now to assign the size traits for orcs that don't have them on_yearly_pulse
character_event = {
	id = DR_greenskins.1
	title = YOUR_SIZE
	desc = EVTDESCDR_greenskins.1 # Orcs and goblins grow in size from the number of battles they have fought in and won. As you have reached the age of adulthood, your size is...
	picture = "GFX_evt_orcs"
	min_age = 16
	is_triggered_only = yes

	trigger = {
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		nor = {
			trait = orc_small
			trait = orc_average
			trait = orc_large
			trait = orc_huge
			trait = goblin_small
			trait = goblin_average
			trait = goblin_large
			trait = goblin_huge
		}
	}

	immediate = {
		log = "work event DR_greenskins.1"
	}

	option = {
		name = ARRRRGH
		if = {
			limit = {
				culture_group = orc_group
				martial = 5
				NOT = {
					martial = 10
				}
			}
			add_trait = orc_small
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 10
				NOT = {
					martial = 15
				}
			}
			add_trait = orc_average
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 15
				NOT = {
					martial = 25
				}
			}
			add_trait = orc_large
		}
		if = {
			limit = {
				culture_group = orc_group
				martial = 25
			}
			add_trait = orc_huge
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 5
				NOT = {
					martial = 10
				}
			}
			add_trait = goblin_small
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 10
				NOT = {
					martial = 15
				}
			}
			add_trait = goblin_average
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 15
				NOT = {
					martial = 25
				}
			}
			add_trait = goblin_large
		}
		if = {
			limit = {
				culture_group = goblin_group
				martial = 25
			}
			add_trait = goblin_huge
		}
	}
}

# Event for greenskins to grow in size from battles
character_event = {
	id = DR_greenskins.2
	desc = EVTDESCDR_greenskins.2 # With each victory in battle, you grow a little in size and strength!
	picture = "GFX_evt_orcs"
	is_triggered_only = yes

	trigger = {
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		nor = {
			trait = orc_huge
			trait = goblin_huge
		}
	}

	weight_multiplier = {
		factor = 1
		modifier = {
			factor = 2
			trait = brave
		}
		modifier = {
			factor = 2
			trait = cruel
		}
		modifier = {
			factor = 2
			trait = wroth
		}
	}

	immediate = {
		log = "work event DR_greenskins.2"
	}

	option = {
		name = "EVTOPTAORCSGROW2"
		if = {
			limit = {
				trait = orc_small
			}
			add_trait = orc_average
			remove_trait = orc_small
		}
		if = {
			limit = {
				trait = orc_average
			}
			add_trait = orc_large
			remove_trait = orc_average
		}
		if = {
			limit = {
				trait = orc_large
			}
			add_trait = orc_huge
			remove_trait = orc_large
		}
		if = {
			limit = {
				trait = goblin_small
			}
			add_trait = goblin_average
			remove_trait = goblin_small
		}
		if = {
			limit = {
				trait = goblin_average
			}
			add_trait = goblin_large
			remove_trait = goblin_average
		}
		if = {
			limit = {
				trait = goblin_large
			}
			add_trait = goblin_huge
			remove_trait = goblin_large
		}
	}
}

# This event chain determines who gets to become shamans. For now, it is based on the mystic trait
# Landed characters only so far, as studying magic is expensive and we don't want all greenskin mystics to become shamans on_bi_yearly_pulse
character_event = {
	id = DR_greenskins.100
	title = MAGIC
	desc = greenskinmagic0 # You iz becoming weird as youz mates say... but maybe you iz just becoming mo' powerful! Do you want to start studying the deeper mysteries of Gork and Mork?
	picture = "GFX_evt_orcs"
	min_age = 16
	religion_group = greenskin_gods_group
	is_triggered_only = yes

	trigger = {
		is_landed = yes
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		OR = {
			piety = 1000
			trait = scholar
			trait = mystic
		}
		nor = {
			trait = lore_waaagh
			has_character_flag = refused_shaman
		}
	}

	immediate = {
		log = "work event DR_greenskins.100"
	}

	option = {
		name = greenskinmagic0A # I will get da power!
		trigger = {
			nor = {
				trait = craven
				trait = content
			}
		}
		add_trait = lore_waaagh
		z_magic_power_sequential_addition = yes
		top_liege = {
			if = {
				limit = {
					OR = {
						culture_group = orc_group
						culture_group = goblin_group
					}
				}
			}
			character_event = {
				id = DR_greenskins.102 days = 7
			}
		}
		father = {
			character_event = {
				id = DR_greenskins.101
			}
		}
		mother = {
			character_event = {
				id = DR_greenskins.101
			}
		}
	}

	option = {
		name = greenskinmagic0B # Nah, I just likes to use my smasher!
		trigger = {
			trait = craven
			trait = content
		}
		set_character_flag = refused_shaman
	}
}

	}

			has_character_modifier = weak_warboss

			has_character_flag = weak_warboss_bashed_heads
		}

		# culture_group = goblin_group

			name = weak_warboss
			duration = -1
		}
	}
}

# Black Orc shows up at court and challenges too weak or too successful Greenskin ruler
character_event = {
	id = DR_greenskins.210
	desc = DR_greenskins210DESC
	picture = GFX_evt_greenskin_black_orc
	capable_only = yes
	only_men = yes
	only_playable = yes
	religion_group = greenskin_gods_group

	mean_time_to_happen = {
		months = 600
	}

	trigger = {
		is_ruler = yes
		OR = {
			has_character_modifier = weak_warboss
			realm_size = 20
		}
		NOT = {
			trait = black_orc_trait
		}
	}

	immediate = {
		log = "work event DR_greenskins.210"
	}

	option = {
		name = DR_greenskins210OPTA
		create_random_soldier = {
			random_traits = yes
			dynasty = culture
			culture = orc_main
			religion = greenskin_gods
			age = 30
			attributes = {
				martial = 15
				stewardship = 5
			}
			trait = brilliant_strategist
			trait = orc_huge
		}
		new_character = {
			add_trait = black_orc_trait
			set_graphical_culture = orc_black
			random_list = {
				50 = {
					add_trait = greenskin_gods_gork
				}
				50 = {
					add_trait = greenskin_gods_mork
				}
			}
			add_character_modifier = {
				modifier = recent_duel_timer
				duration = 720
				hidden = yes
			}
			save_event_target_as = duelist_a
			set_character_flag = duel
		}
		ROOT = {
			save_event_target_as = duelist_b
			set_character_flag = duel
			hidden_tooltip = {
				character_event = {
					id = duelengine.1
				}
			}
		}
	}
}

# Greenskin province culture flips if ruler is of a Greenskin subculture in its same group
province_event = {
	id = DR_greenskins.211
	desc = DR_greenskins211DESC
	picture = GFX_evt_orc_banner
	religion_group = greenskin_gods_group

	mean_time_to_happen = {
		days = 10
	}

	trigger = {
		OR = {
			culture_group = goblin_group
			culture_group = orc_group
		}
		OR = {
			AND = {
				OR = {
					culture = orc_main
					culture = orc_savage
				}
				holder_scope = {
					culture_group = orc_group
					NOT = {
						culture = ROOT
					}
				}
			}
			AND = {
				OR = {
					culture = goblin_main
					culture = goblin_night
					culture = goblin_forest
				}
				holder_scope = {
					culture_group = goblin_group
					NOT = {
						culture = ROOT
					}
				}
			}
		}
	}

	immediate = {
		log = "work event DR_greenskins.211"
	}

	option = {
		name = DR_greenskins211OPTA
		holder_scope = {
			ROOT = {
				religion = PREV
				culture = PREV
			}
		}
	}
}
