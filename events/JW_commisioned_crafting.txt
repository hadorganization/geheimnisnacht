# 0 Triggered by decision. Set the event chain targets

# 1-7 Crafting menus

# 11 Commission rejected notification

# 30-34 Commission a sword

# 35-39 Sword finished

namespace = commissioned_crafting

# Bounce event. Set the buyer and wizard
character_event = {
	id = commissioned_crafting.0
	hide_window = yes
	# Triggered by targetted decision
	is_triggered_only = yes

	immediate = {
		FROM = {
			save_event_target_as = wizard
		}
		ROOT = {
			save_event_target_as = buyer
			character_event = {
				id = commissioned_crafting.1
			}
		}
	}
}

character_event = {
	id = commissioned_crafting.54
	title = commissioned_crafting_title.53
	desc = commissioned_crafting_desc.54
	picture = GFX_evt_lab
	is_triggered_only = yes

	option = {
		name = EXCELLENT
		event_target:wizard = {
			remove_trait = crafting_magic_item
		}
		if = {
			limit = {
				trait = pretty
			}
			add_trait = beautiful
			hidden_tooltip = {
				remove_trait = pretty
			}
			break = yes
		}
		if = {
			limit = {
				trait = fair
			}
			add_trait = pretty
			hidden_tooltip = {
				remove_trait = fair
			}
			break = yes
		}
		if = {
			limit = {
				NOT = {
					trait = beautiful
				}
				NOT = {
					trait = pretty
				}
				NOT = {
					trait = fair
				}
				NOT = {
					trait = plain
				}
				NOT = {
					trait = ugly
				}
				NOT = {
					trait = unsightly
				}
				NOT = {
					trait = monstrous
				}
			}
			add_trait = fair
			break = yes
		}
		if = {
			limit = {
				trait = plain
			}
			remove_trait = plain
			break = yes
		}
		if = {
			limit = {
				trait = ugly
			}
			add_trait = plain
			hidden_tooltip = {
				remove_trait = ugly
			}
			break = yes
		}
		if = {
			limit = {
