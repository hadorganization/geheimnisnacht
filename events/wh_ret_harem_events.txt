# Becomes proud
character_event = {
	id = haremsins.003
	desc = "EVTDESCharemsins003"
	picture = "GFX_evt_sultan"
	ai = no
	prisoner = no
	only_playable = yes

	immediate = {
		log = "work event id = haremsins.003 for [Root.GetBestName]"
		set_character_flag = harem_event_happened
	}

	trigger = {
		trait = retired_with_harem
		OR = {
			NOT = {
				has_character_flag = harem_event_happened
			}
			had_character_flag = {
				flag = harem_event_happened
				days = 5
			}
		}
		NOT = {
			trait = proud
		}
	}

	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 0.75
			NOT = {
				has_character_flag = harem_event_happened
			}
		}
		modifier = {
			factor = 1.5
			has_character_flag = harem_event_happened
		}
		modifier = {
			factor = 2
			trait = humble
		}
		modifier = {
			factor = 2
			trait = shy
		}
		modifier = {
			factor = 0.75
			trait = gregarious
		}
		modifier = {
			factor = 0.5
			trait = proud
		}
		modifier = {
			factor = 0.75
			trait = fair
		}
		modifier = {
			factor = 1.25
			trait = quick
		}
		modifier = {
			factor = 1.5
			trait = genius
		}
	}

	option = {
		name = EVTOPTAharemsins003
		add_trait = proud
		piety = 25
		prestige = 25
		if = {
			limit = {
				trait = depressed
			}
			remove_trait = depressed
		}
	}
}
