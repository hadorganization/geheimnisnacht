# on-birth event to give ormazdic sayyid/mirza traits
character_event = {
	id = ormazdian.0
	desc = ormazdian0desc
	picture = GFX_evt_bishop
	is_triggered_only = yes

	trigger = {
		NOR = {
			trait = ormazd_sayyid
			trait = non_ormazd_sayyid
			trait = ormazd_mirza
			trait = non_ormazd_mirza
		}
		OR = {
			father_even_if_dead = {
				trait = ormazd_sayyid
			}
			mother_even_if_dead = {
				trait = ormazd_sayyid
			}
			father_even_if_dead = {
				trait = non_ormazd_sayyid
			}
			mother_even_if_dead = {
				trait = non_ormazd_sayyid
			}
		}
	}

	immediate = {
		log = "work event id = ormazdian.0 for [Root.GetBestName]"
	}

	option = {
		name = OK
		if = {
			limit = {
				NOT = {
					religion = ormazd
				}
				OR = {
					father_even_if_dead = {
						trait = ormazd_sayyid
					}
					father_even_if_dead = {
						trait = non_ormazd_sayyid
					}
				}
			}
			add_trait = non_ormazd_sayyid
			break = yes
		}
		if = {
			limit = {
				NOT = {
					religion = ormazd
				}
				OR = {
					mother_even_if_dead = {
						trait = ormazd_sayyid
					}
					mother_even_if_dead = {
						trait = non_ormazd_sayyid
					}
				}
			}
			add_trait = non_ormazd_mirza
			break = yes
		}
		if = {
			limit = {
				religion = ormazd
				OR = {
					father_even_if_dead = {
						trait = non_ormazd_sayyid
					}
					father_even_if_dead = {
						trait = ormazd_sayyid
					}
				}
			}
			add_trait = ormazd_sayyid
			break = yes
		}
		if = {
			limit = {
				religion = ormazd
				OR = {
					mother_even_if_dead = {
						trait = ormazd_sayyid
					}
					mother_even_if_dead = {
						trait = non_ormazd_sayyid
					}
				}
			}
			add_trait = ormazd_sayyid
			break = yes
		}
	}
}

# Maintenance event swapping sayyid/mirza on conversion
character_event = {
	id = ormazdian.1
	desc = ormazdian1desc
	picture = GFX_evt_bishop
	is_triggered_only = yes

	trigger = {
		OR = {
			trait = non_ormazd_sayyid
			trait = ormazd_sayyid
			trait = non_ormazd_mirza
			trait = ormazd_mirza
		}
	}

	immediate = {
		log = "work event id = ormazdian.1 for [Root.GetBestName]"
	}

	option = {
		name = OK
		if = {
			limit = {
				religion = ormazd
			}
			if = {
				limit = {
					trait = non_ormazd_sayyid
				}
				add_trait = ormazd_sayyid
				remove_trait = non_ormazd_sayyid
			}
			if = {
				limit = {
					trait = non_ormazd_mirza
				}
				add_trait = ormazd_mirza
				remove_trait = non_ormazd_mirza
			}
		}
		if = {
			limit = {
				NOT = {
					religion = ormazd
				}
			}
			if = {
				limit = {
					trait = non_ormazd_sayyid
				}
				remove_trait = ormazd_sayyid
				add_trait = non_ormazd_sayyid
			}
			if = {
				limit = {
					trait = non_ormazd_mirza
				}
				remove_trait = ormazd_mirza
				add_trait = non_ormazd_mirza
			}
		}
	}
}

# 7: Destroy caliphate when it can't be inherited by a Sayyid

	id = ormazdian.7
	desc = ormazdian7desc
	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion
	is_triggered_only = yes

	trigger = {
		k_caliph_al_umma = {
			holder_scope = {
				character = ROOT
			}
		}
		NOT = {
			trait = ormazd_sayyid
		}
	}

	immediate = {
		log = "work event id = ormazdian.7 for [Root.GetBestName]"
	}

	option = {
		trigger = {
			OR = {
				any_close_relative = {
					religion = ormazd
					trait = ormazd_sayyid
				}
				any_courtier = {
					religion = ormazd
					trait = ormazd_sayyid
				}
			}
		}
		name = OK
		if = {
			limit = {
				any_close_relative = {
					religion = ormazd
					trait = ormazd_sayyid
				}
			}
			k_caliph_al_umma = {
				ROOT = {
					random_dynasty_member = {
						limit = {
							religion = ormazd
							trait = ormazd_sayyid
						}
						PREV = {
							grant_title = PREVPREVPREV
						}
					}
				}
			}
		}
		if = {
			limit = {
				any_courtier = {
					religion = ormazd
					trait = ormazd_sayyid
				}
				NOT = {
					any_close_relative = {
						religion = ormazd
						trait = ormazd_sayyid
					}
				}
			}
			k_caliph_al_umma = {
				ROOT = {
					random_courtier = {
						limit = {
							religion = ormazd
							trait = ormazd_sayyid
						}
						PREV = {
							grant_title = PREVPREVPREV
						}
					}
				}
			}
		}
	}

	option = {
		name = OK
		trigger = {
			NOT = {
				OR = {
					any_courtier = {
						religion = ormazd
						trait = ormazd_sayyid
					}
					any_close_relative = {
						religion = ormazd
						trait = ormazd_sayyid
					}
				}
			}
		}
		k_caliph_al_umma = {
			unsafe_destroy_landed_title = THIS
			ROOT = {
				narrative_event = {
					id = ormazdian.9
				}
			}
		}
	}
}

# 8: Trigger for 7 (triggered by on_death)

	id = ormazdian.8
	desc = ormazdian8desc
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		k_caliph_al_umma = {
			holder_scope = {
				character = ROOT
			}
		}
	}

	immediate = {
		log = "work event id = ormazdian.8 for [Root.GetBestName]"
	}

	option = {
		name = OK
		current_heir = {
			character_event = {
				id = ormazdian.7
			}
		}
	}
}
