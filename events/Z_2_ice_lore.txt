namespace = ice_lore

character_event = {
	id = ice_lore.101# Drain Magic Power 1"
	desc = ice_lore.101_desc# "Your spell burns through the air, taking from all enemies the ability to cast. "
	is_triggered_only = yes
	picture = GFX_evt_mage_lore_necromancy

	option = {
		name = ice_lore.101a# No
		character_event = {
			id = z_spell_menu.99996
		}
		event_target:magic_caster = {
			set_variable = {
				which = zdmg value = 5
			}
			set_variable = {
				which = zmoraledmg value = 10
			}
			change_variable = {
				which = zdmg which = caster_level
			}
			change_variable = {
				which = zmoraledmg which = caster_level
			}
		}
		any_character = {
			limit = {
				war_with = root
				any_army = {
					location = {
						province = event_target:spellloc
					}
				}
			}
			random_army = {
				limit = {
					location = {
						province = event_target:spellloc
					}
				}
				leader = {
					magic_points_setup = yes
					save_event_target_as = spelltarget
					magic_resist_test = yes
				}
				if = {
					limit = {
						event_target:magic_caster = {
							has_character_flag = resisted
						}
					}
					event_target:magic_caster = {
						clr_character_flag = resisted
						divide_variable = {
							which = zdmg value = 2
						}
						divide_variable = {
							which = zmoraledmg value = 2
						}
					}
				}
				magic_dmg_effect = yes
				morale_dmg_effect = yes
			}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					at_location = event_target:spellloc
				}
			}
			siege = {
				enemy = {
					magic_dmg_effect = yes
					morale_dmg_effect = yes
				}
			}
		}
	}
}

character_event = {
	id = ice_lore.402# Soul Quench 1"
	desc = ice_lore.402# "Your power consumes the life of your chosen target "
	is_triggered_only = yes
	hide_window = yes
	picture = GFX_evt_mage_lore_necromancy

	immediate = {
		if = {
			limit = {
				event_target:magic_caster = {
					check_variable = {
						which = "icestorm" value = 0
					}
				}
			}
			event_target:magic_caster = {
				change_variable = {
					which = "icestorm" value = -1
				}
			}
			event_target:magic_caster = {
				set_variable = {
					which = zdmg value = 1
				}
				set_variable = {
					which = zmoraledmg value = 5
				}
				change_variable = {
					which = zdmg which = caster_level
				}
				change_variable = {
					which = zmoraledmg which = caster_level
				}
			}
			any_character = {
				limit = {
					war_with = root
					any_army = {
						location = {
							province = event_target:spellloc
						}
					}
				}
				random_army = {
					limit = {
						location = {
							province = event_target:spellloc
						}
					}
					leader = {
						magic_points_setup = yes
						save_event_target_as = spelltarget
						magic_resist_test = yes
					}
					if = {
						limit = {
							event_target:magic_caster = {
								has_character_flag = resisted
							}
						}
						event_target:magic_caster = {
							clr_character_flag = resisted
							divide_variable = {
								which = zdmg value = 2
							}
							divide_variable = {
								which = zmoraledmg value = 2
							}
						}
					}
					magic_dmg_effect = yes
					morale_dmg_effect = yes
				}
			}
			any_character = {
				if = {
					limit = {
						or = {
							is_vassal_or_below = root
							character = root
						}
						at_location = event_target:spellloc
					}
					siege = {
						enemy = {
							magic_dmg_effect = yes
							morale_dmg_effect = yes
						}
					}
					random = {
						chance = 1
						event_target:spellloc = {
							random_province_holding = {
								destroy_settlement = this
							}
						}
					}
				}
			}
			repeat_event = {
				id = high_magic_lore.402 days = 1
			}
		}
	}
}
