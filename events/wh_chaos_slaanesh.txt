# on_yearly_pulse
character_event = {
	id = wh_chaos_slaanesh.4
	hide_window = yes
	only_playable = yes
	only_capable = yes
	prisoner = no
	min_age = 16
	religion = slaanesh
	is_triggered_only = yes

	trigger = {
		NOT = {
			has_character_flag = Pet_Daemonette_helped_ambition
		}
		any_friend = {
			trait = slaanesh_daemonette
		}
		OR = {
			has_ambition = obj_slaanesh_denounce_celibacy
			has_ambition = obj_slaanesh_indulge_sexual_excess
			has_ambition = obj_slaanesh_embrace_cruelty
			has_ambition = obj_slaanesh_intemperance_food
			has_ambition = obj_slaanesh_indulge_greed
			has_ambition = obj_slaanesh_reject_hate
			has_ambition = obj_slaanesh_embrace_pride
			has_ambition = obj_slaanesh_purge_imperfection
		}
	}

	immediate = {
		log = "work event wh_chaos_slaanesh.4"
		set_character_flag = Pet_Daemonette_helped_ambition
		any_friend = {
			limit = {
				trait = slaanesh_daemonette
			}
			character_event = {
				id = wh_chaos_slaanesh.5
			}
		}
	}
}

# Pet_Daemonette bounce event
character_event = {
	id = wh_chaos_slaanesh.5
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		log = "work event wh_chaos_slaanesh.5"
		if = {
			limit = {
				OR = {
					FROM = {
						or = {
							trait = zealous
							trait = ambitious
						}
					}
					trait = genius
				}
			}
			random = {
				chance = 75
				FROM = {
					set_character_flag = slaanesh_obj_success
				}
			}
		}
		if = {
			limit = {
				FROM = {
					nor = {
						trait = zealous
						trait = ambitious
					}
				}
				NOT = {
					trait = genius
				}
			}
			random = {
				chance = 25
				FROM = {
					set_character_flag = slaanesh_obj_success
				}
			}
		}
		FROM = {
			character_event = {
				id = wh_chaos_slaanesh.6
			}
		}
	}
}

character_event = {
	id = wh_chaos_slaanesh.6
	desc = slaaneshbless6
	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion
	is_triggered_only = yes

	immediate = {
		log = "work event wh_chaos_slaanesh.6"
	}

	option = {
		name = slaaneshbless6A
		trigger = {
			has_character_flag = slaanesh_obj_success
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_denounce_celibacy
				trait = celibate
			}

		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_indulge_sexual_excess
				trait = chaste
			}
			remove_trait = chaste
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_embrace_cruelty
				trait = shy
			}
			remove_trait = shy
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_embrace_cruelty
				trait = kind
			}
			remove_trait = kind
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_intemperance_food
				trait = temperate
			}
			remove_trait = temperate
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_indulge_greed
				trait = charitable
			}
			remove_trait = charitable
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_reject_hate
				trait = wroth
			}
			remove_trait = wroth
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_embrace_pride
				trait = humble
			}
			remove_trait = humble
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_purge_imperfection
				trait = disfigured
			}
			remove_trait = disfigured
		}
		if = {
			limit = {
				has_ambition = obj_slaanesh_purge_imperfection
				trait = ugly
			}
			remove_trait = ugly
		}
		clr_character_flag = slaanesh_obj_success
	}

	option = {
		name = slaaneshbless6B
		trigger = {
			NOT = {
				has_character_flag = slaanesh_obj_success
			}
		}
		if = {
			limit = {
				NOT = {
					trait = diligent
				}
			}
			prestige = -10
			piety = -10
			random = {
				chance = 5
				add_trait = diligent
			}
		}
	}
