# Magritta Re-Established on_yearly_pulse
character_event = {
	id = DR_misc.200
	hide_window = yes
	religion = myrmidian
	only_playable = yes
	is_triggered_only = yes

	trigger = {
		has_landed_title = b_inquisitoria_general
		is_theocracy = yes
		NOT = {
			has_landed_title = d_high_temple_of_magritta
		}
		OR = {
			independent = yes
			top_liege = {
				higher_real_tier_than = DUKE
				religion = ROOT
			}
		}
	}

	immediate = {
		log = "work event DR_misc.200"
		any_liege = {
			limit = {
				tier = EMPEROR
			}
			save_event_target_as = new_liege
		}
		any_liege = {
			limit = {
				tier = KING
			}
			save_event_target_as = new_liege
		}
		d_high_temple_of_magritta = {
			grant_title = ROOT
		}
		set_defacto_liege = event_target:new_liege
		if = {
			limit = {
				k_cult_of_myrmidia = {
					holder_scope = {
						NOT = {
							has_landed_title = d_high_temple_of_magritta
						}
						NOT = {
							has_landed_title = d_high_temple_of_remas
						}
					}
				}
			}
			any_liege = {
				limit = {
					tier = EMPEROR
				}
				save_event_target_as = new_liege
			}
			k_cult_of_myrmidia = {
				holder_scope = {
					grant_title = d_high_temple_of_magritta
				}
			}
			k_cult_of_myrmidia = {
				holder_scope = {
					grant_title = b_inquisitoria_general
				}
			}
			set_defacto_liege = event_target:new_liege
		}
	}
}
