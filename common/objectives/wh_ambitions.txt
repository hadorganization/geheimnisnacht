		is_landed = yes
		is_incapacitated_trigger = no
		war = no
	}
	chance = {
		factor = 100
		modifier = {
			factor = 100 # Don't discard objective if there is war
			war = yes
		}
		modifier = {
			factor = 0 # Only independent AIs
			independent = no
		}
		modifier = {
			factor = 0
			trait = content
		}
		modifier = {
			factor = 1.5
			trait = ambitious
		}
		modifier = {
			factor = 0
			trait = craven
		}
		modifier = {
			factor = 2
			trait = brave
		}
		modifier = {
			factor = 0.5
			not = {
				martial = 8
			}
		}
		modifier = {
			factor = 0.25
			not = {
				martial = 5
			}
		}
		modifier = {
			factor = 0
			not = {
				martial = 3
			}
		}
		modifier = {
			factor = 1.25
			martial = 10
		}
		modifier = {
			factor = 1.25
			martial = 13
		}
		modifier = {
			factor = 1.25
			martial = 16
		}
		modifier = {
			factor = 1.25
			martial = 19
		}
		modifier = {
			factor = 1.25
			martial = 22
		}
	}
	success = {
		custom_tooltip = {
			text = DR_obj_win_war
			hidden_tooltip = {
				has_character_flag = obj_win_war_won_war
			}
		}
	}
	abort = {
		or = {
			is_ruler = no
			is_incapacitated_trigger = yes
		}
	}
	abort_effect = {
	}
	effect = {
		if = {
			limit = {
				not = {
					has_character_flag = obj_win_war
				}
			}
			change_martial = 1
			prestige = 50
		}
		if = {
			limit = {
				has_character_flag = obj_win_war
			}
			prestige = 25
		}
		set_character_flag = obj_win_war
		clr_character_flag = obj_win_war_won_war
	}
}
