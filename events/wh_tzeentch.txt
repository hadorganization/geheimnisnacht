namespace = fickletzeentch

# The Master of Fortune
character_event = {
	id = fickletzeentch.0
	title = Inspiration
	desc = fickletzeentch # The great Architect of Fate has touched your addled brain, giving you brief glimpses of inspiration and clarity.
	picture = GFX_evt_library
	min_age = 16
	capable_only = yes
	prisoner = no
	is_triggered_only = yes

	trigger = {
		is_landed = yes
		religion = tzeentch
		NOT = {
			trait = genius
		}
		NOT = {
			trait = brilliant
		}
		NOT = {
			trait = quick
		}
		NOT = {
			trait = smart
		}
		NOT = {
			trait = bright
		}
		NOT = {
			learning = 8
		}
		NOT = {
			has_character_flag = touched_by_tzeentch
		}
		NOT = {
			has_character_flag = failed_tzeentch_test
		}
		NOT = {
			has_character_flag = refused_dark
		}
		NOT = {
			trait = lore_dark_magic
		}
		NOT = {
			trait = creature_beastman
		} # only on_birth
	}

	option = {
		name = TestofTzeentch9A # I will carefully study these visions...
		random_list = {
			75 = {
				if = {
					limit = {
						NOT = {
							trait = smart
						}
					}
					add_trait = smart
					hidden_tooltip = {
						remove_trait = dim
						remove_trait = dumb
						remove_trait = slow
						remove_trait = idiot
						remove_trait = imbecile
						change_learning = 1
						change_intrigue = 1
					}
				}
				hidden_tooltip = {
					character_event = {
						id = fickletzeentch.1
					}
				}
			}
			25 = {
				if = {
					limit = {
						NOT = {
							trait = mut_thirdeye
						}
						NOT = {
							trait = mutation_minor
						}
						NOT = {
							trait = dumb
						}
					}
					add_trait = mutation_minor
					add_trait = dumb
					set_character_flag = mut_thirdeye
				}
				hidden_tooltip = {
					change_variable = {
						which = "mutation_counter" value = 1
					}
					character_event = {
						id = mutation.2
					}
					set_character_flag = failed_tzeentch_test
				}
			}
		}
	}

	option = {
		name = TestofTzeentch9B # I would risk everything, to know the infinite!
		random_list = {
			55 = {
				if = {
					limit = {
						NOT = {
							trait = brilliant
						}
					}
					add_trait = brilliant
					hidden_tooltip = {
						remove_trait = dim
						remove_trait = dumb
						remove_trait = slow
						remove_trait = idiot
						remove_trait = imbecile
					}
				}
				hidden_tooltip = {
					character_event = {
						id = fickletzeentch.1
					}
					change_learning = 2
					change_intrigue = 2
				}
			}
			45 = {
				if = {
					limit = {
						NOT = {
							trait = mut_tongue
						}
						NOT = {
							trait = mutation_major
						}
						NOT = {
							trait = imbecile
						}
						NOT = {
							trait = disfigured
						}
					}
					add_trait = disfigured
					add_trait = imbecile
					add_trait = mutation_major
					set_character_flag = mut_tongue
					hidden_tooltip = {
						remove_trait = genius
						remove_trait = brilliant
						remove_trait = quick
						remove_trait = smart
						remove_trait = bright
					}
				}
				hidden_tooltip = {
					change_variable = {
						which = "mutation_counter" value = 5
					}
					character_event = {
						id = mutation.1001
					}
					set_character_flag = failed_tzeentch_test
				}
			}
		}
	}

	option = {
		name = TestofTzeentch9C # I fear the power of change...
		if = {
			limit = {
				NOT = {
					trait = content
				}
				NOT = {
					trait = craven
				}
			}
			add_trait = craven
			add_trait = content
			hidden_tooltip = {
				remove_trait = ambitious
				remove_trait = brave
			}
			set_character_flag = refused_dark
		}
	}
}
