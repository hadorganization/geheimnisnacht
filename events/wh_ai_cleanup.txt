# on death to pass an amazon barony to a randomly generated female

	id = aicleanup.001
	hide_window = yes
	religion = amazon
	culture = amazon_main

	max_age = 100
	is_triggered_only = yes

		primary_title = {

		log = "work event id = aicleanup.001 for [Root.GetBestName]"

			25 = {
				create_character = {
					random_traits = yes
					dynasty = NONE
					religion = ROOT
					culture = ROOT
					female = yes
					age = 20
				}
			}
			25 = {
				create_character = {
					random_traits = yes
					dynasty = RANDOM
					religion = ROOT
					culture = ROOT
					female = yes
					age = 30
				}
			}
			25 = {
				create_character = {
					random_traits = yes
					dynasty = RANDOM
					religion = ROOT
					culture = ROOT
					female = yes
					age = 40
				}
			}

				create_character = {

# Everqueen Trait Lost or gain on_yearly_pulse on_new_holder
character_event = {
	id = aicleanup.005
	hide_window = yes
	religion = cadai
	only_women = yes
	is_triggered_only = yes

	trigger = {
		or = {
			and = {
				trait = everqueen
				NOT = {
					has_landed_title = k_avelorn
				}
			}
			and = {			
				trait = everqueen_heir
				has_landed_title = k_avelorn
			}
		}
	}

	immediate = {
		log = "work event id = aicleanup.005 for [Root.GetBestName]"
		remove_trait = everqueen		
		# fix everchild

			limit = {
				trait = everqueen_heir
				has_landed_title = k_avelorn
			}
			remove_trait = everqueen_heir
			add_trait = everqueen
		}
	}
}

	is_triggered_only = yes
	only_rulers = yes

		is_feudal = yes

	}

					culture_group = druchi_group

			}

	}

# Give the proper age traits to dwarves

	id = aicleanup.023
	hide_window = yes
	only_men = yes
	religion_group = dwarf_gods_group
	is_triggered_only = yes

	trigger = {
		trait = creature_dwarf
		NOT = {
			trait = dwarf_slayer
		}
	}

	immediate = {
		log = "work event id = aicleanup.023 for [Root.GetBestName]"
		if = {
			limit = {
				NOT = {
					age = 75
				}
			}
			add_trait = beardling
		}
		if = {
			limit = {
				age = 75
				martial = 20
				nor = {
					trait = quarreller
					trait = hammerer
					trait = dwarf_warrior
					trait = thunderer
				}
			}
			add_trait = thunderer
			remove_trait = beardling
		}
		if = {
			limit = {
				age = 75
				diplomacy = 15
				nor = {
					trait = quarreller
					trait = hammerer
					trait = dwarf_warrior
					trait = thunderer
				}
			}
			add_trait = quarreller
			remove_trait = beardling
		}
		if = {
			limit = {
				age = 75
				nor = {
					martial = 20
					diplomacy = 15
					trait = quarreller
					trait = hammerer
					trait = dwarf_warrior
					trait = thunderer
				}
			}
			add_trait = dwarf_warrior
			remove_trait = beardling
		}
		if = {
			limit = {
				age = 250
				nor = {
					trait = longbeard
					trait = ironbreaker
				}
			}
			add_trait = longbeard
			remove_trait = beardling
		}
	}
}
