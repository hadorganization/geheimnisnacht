# Death system for long-lived beings
# Elves, Dwarfs, Chaos Dwarfs
#
# Known elves lifespan
# 602, 381, 649 (Phoenix kings - so more longlived let's say 1.5)
## Bel Shanaar reigned for 1669 years
## & Bel Khorandis reigned for 1189 years
## This implies they were actually much older, as the phoenix king is an elected title
## Shadowblade was considered "very young" at 150 years old.
## I've seen speculation in several locations that the average lifespan is 2500
## we may want to address this accordingly -Erik
# let's say then that the average elves lifespan is ~1500 years
#
character_event = {
    id = racial.2
    desc = "EVTDESC_RACIAL_2"
    picture = "GFX_evt_recovery"

    min_age = 200
    trigger = {
			NOT = { trait = avatar_isha }
			NOT = { trait = avatar_kurnous }
        AND = {
		OR = {
            trait = creature_elf
            trait = creature_chaos_dwarf
            trait = creature_dwarf
			}
			NOT = { trait = malekith }
			OR = { NOT = { trait = immortal }
				trait = incapable }
        }
    }
	
    mean_time_to_happen = {
        years = 100
        # first define the expected lifespan
        modifier = {
            factor = 16
            trait = creature_elf
        }
        modifier = {
            factor = 100
            trait = creature_elf
            NOT = { age = 950 }
        }
        modifier = {
            factor = 5
            OR = { mother = { is_alive = yes }
		  father = { is_alive = yes }
		 }
        }
        modifier = {
            factor = 4
            trait = creature_chaos_dwarf
        }
        modifier = {
            factor = 4
            trait = creature_dwarf
        }
        # now health modifiers
        modifier = {
            factor = 0.6
            NOT = { health = 25.1 }
        }
        modifier = {
            factor = 0.6
            NOT = { health = 27 }
        }
        modifier = {
            factor = 0.75
            NOT = { health = 28 }
        }
        modifier = {
            factor = 0.8
            NOT = { health = 29 }
        }
        modifier = {
            factor = 0.8
            NOT = { health = 30 }
        }
        modifier = {
            factor = 1.1
            health = 30.5
        }
        modifier = {
            factor = 1.2
            health = 31
        }
        modifier = {
            factor = 1.2
            health = 32
        }
        modifier = {
            factor = 1.2
            health = 33
        }
        modifier = {
            factor = 0.1
            trait = incapable
        }
        modifier = {
            factor = 10
            trait = avatar_isha
        }
        modifier = {
            factor = 10
            trait = avatar_kurnous
        }

    }
    option = {
        name = "EVTOPT_RACIAL_2A"
		if = {
 limit = {
 is_invincible_trigger = yes
 }

			antideath_effect = yes
			break = yes 
}
death = { death_reason = death_natural }
    }
}
